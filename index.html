// Firebase init const firebaseConfig = { apiKey: "ТВІЙ_API_KEY", authDomain: "ТВІЙ_AUTH_DOMAIN", projectId: "ТВІЙ_PROJECT_ID", storageBucket: "ТВІЙ_STORAGE_BUCKET", messagingSenderId: "ТВІЙ_SENDER_ID", appId: "ТВІЙ_APP_ID" }; firebase.initializeApp(firebaseConfig); const auth = firebase.auth(); const db = firebase.firestore(); let currentUser = null; let currentBook = null; // ================= AUTH ================= function signup(){ auth.createUserWithEmailAndPassword(email.value, password.value) .then(() => alert("Акаунт створено ✔")) .catch(e => alert("Помилка: " + e.message)); } function signin(){ auth.signInWithEmailAndPassword(email.value, password.value) .then(() => console.log("Logged in")) .catch(e => alert("Помилка: " + e.message)); } function logout(){ auth.signOut(); } // ================= STATE ================= auth.onAuthStateChanged(user=>{ if(user){ currentUser = user; login.style.display="none"; app.style.display="flex"; loadBooks(); }else{ login.style.display="block"; app.style.display="none"; } }); // ================= BOOKS ================= function addBook(){ if(!bookInput.value) return; db.collection("users") .doc(currentUser.uid) .collection("books") .add({name: bookInput.value}); bookInput.value=""; } function loadBooks(){ db.collection("users") .doc(currentUser.uid) .collection("books") .onSnapshot(snap=>{ bookList.innerHTML=""; snap.forEach(doc=>{ let div=document.createElement("div"); div.className="book"; div.innerText=doc.data().name; div.onclick=()=>openBook(doc.id, doc.data().name); bookList.appendChild(div); }); }); } function openBook(id,name){ currentBook=id; bookTitle.innerText=name; loadNotes(); } // ================= NOTES ================= function addNote(){ if(!currentBook || !noteInput.value) return; db.collection("users") .doc(currentUser.uid) .collection("books") .doc(currentBook) .collection("notes") .add({text: noteInput.value}); noteInput.value=""; } function loadNotes(){ db.collection("users") .doc(currentUser.uid) .collection("books") .doc(currentBook) .collection("notes") .onSnapshot(snap=>{ notes.innerHTML=""; snap.forEach(doc=>{ let div=document.createElement("div"); div.innerText="📌 " + doc.data().text; notes.appendChild(div); }); }); }
